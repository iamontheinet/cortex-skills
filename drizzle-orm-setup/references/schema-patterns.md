# Schema Patterns

Complete reference for defining tables, columns, enums, indexes, and relations in Drizzle ORM.

## Table Definitions

### PostgreSQL

```typescript
import {
  pgTable, pgEnum,
  integer, serial, bigint,
  varchar, text, char,
  boolean,
  timestamp, date, time, interval,
  numeric, real, doublePrecision,
  json, jsonb,
  uuid,
  uniqueIndex, index,
} from 'drizzle-orm/pg-core';

// Custom enum (PostgreSQL only — creates a real DB enum type)
export const userRoleEnum = pgEnum('user_role', ['admin', 'member', 'guest']);

export const users = pgTable('users', {
  id: integer('id').primaryKey().generatedAlwaysAsIdentity(),
  email: varchar('email', { length: 320 }).notNull(),
  name: varchar('name', { length: 100 }).notNull(),
  role: userRoleEnum('role').default('member').notNull(),
  isActive: boolean('is_active').default(true).notNull(),
  metadata: jsonb('metadata').$type<{ preferences: Record<string, unknown> }>(),
  createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
  updatedAt: timestamp('updated_at', { withTimezone: true })
    .defaultNow()
    .notNull()
    .$onUpdateFn(() => new Date()),
}, (table) => [
  uniqueIndex('users_email_idx').on(table.email),
]);
```

### MySQL

```typescript
import {
  mysqlTable, mysqlEnum,
  int, serial, bigint,
  varchar, text, char,
  boolean,
  timestamp, datetime, date,
  decimal, float, double,
  json,
  uniqueIndex, index,
} from 'drizzle-orm/mysql-core';

export const users = mysqlTable('users', {
  id: int('id').primaryKey().autoincrement(),
  email: varchar('email', { length: 320 }).notNull(),
  name: varchar('name', { length: 100 }).notNull(),
  role: mysqlEnum('role', ['admin', 'member', 'guest']).default('member').notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
  updatedAt: timestamp('updated_at')
    .defaultNow()
    .notNull()
    .$onUpdateFn(() => new Date()),
}, (table) => [
  uniqueIndex('users_email_idx').on(table.email),
]);
```

### SQLite

```typescript
import {
  sqliteTable,
  integer, real,
  text, blob,
  uniqueIndex, index,
} from 'drizzle-orm/sqlite-core';

export const users = sqliteTable('users', {
  id: integer('id').primaryKey({ autoIncrement: true }),
  email: text('email').notNull(),
  name: text('name').notNull(),
  role: text('role', { enum: ['admin', 'member', 'guest'] }).default('member').notNull(),
  createdAt: integer('created_at', { mode: 'timestamp' })
    .notNull()
    .$defaultFn(() => new Date()),
  updatedAt: integer('updated_at', { mode: 'timestamp' })
    .notNull()
    .$defaultFn(() => new Date())
    .$onUpdateFn(() => new Date()),
}, (table) => [
  uniqueIndex('users_email_idx').on(table.email),
]);
```

## Shared Timestamps Pattern

Extract reusable timestamp columns to avoid repetition:

```typescript
// PostgreSQL
const timestamps = {
  createdAt: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
  updatedAt: timestamp('updated_at', { withTimezone: true })
    .defaultNow()
    .notNull()
    .$onUpdateFn(() => new Date()),
};

// Use in any table
export const posts = pgTable('posts', {
  id: integer('id').primaryKey().generatedAlwaysAsIdentity(),
  title: varchar('title', { length: 255 }).notNull(),
  ...timestamps,
});
```

## Column Modifiers

```typescript
// Common modifiers (work on all column types)
.notNull()                    // NOT NULL constraint
.default('value')             // Static default
.defaultNow()                 // DEFAULT NOW() (timestamps only)
.$defaultFn(() => crypto.randomUUID())  // JS-level default (runs on insert)
.$onUpdateFn(() => new Date())          // JS-level update (runs on every update)
.unique()                     // UNIQUE constraint
.primaryKey()                 // PRIMARY KEY

// PostgreSQL identity columns (preferred over serial)
.generatedAlwaysAsIdentity()        // GENERATED ALWAYS AS IDENTITY
.generatedByDefaultAsIdentity()     // GENERATED BY DEFAULT AS IDENTITY

// Foreign key inline
.references(() => otherTable.id)                              // basic FK
.references(() => otherTable.id, { onDelete: 'cascade' })     // with cascade
.references(() => otherTable.id, { onDelete: 'set null' })    // set null on delete

// Typed JSON (TypeScript only — no DB enforcement)
.jsonb('metadata').$type<{ key: string; value: number }>()
```

## Indexes

Defined as the third argument to the table constructor:

```typescript
export const posts = pgTable('posts', {
  id: integer('id').primaryKey().generatedAlwaysAsIdentity(),
  title: varchar('title', { length: 255 }).notNull(),
  slug: varchar('slug', { length: 255 }).notNull(),
  authorId: integer('author_id').notNull().references(() => users.id),
  status: varchar('status', { length: 20 }).default('draft').notNull(),
  publishedAt: timestamp('published_at'),
}, (table) => [
  // Unique index
  uniqueIndex('posts_slug_idx').on(table.slug),

  // Composite index
  index('posts_author_status_idx').on(table.authorId, table.status),

  // Partial index (PostgreSQL only)
  index('posts_published_idx')
    .on(table.publishedAt)
    .where(sql`${table.status} = 'published'`),
]);
```

## Composite Primary Keys

```typescript
import { primaryKey } from 'drizzle-orm/pg-core';

export const teamMembers = pgTable('team_members', {
  userId: integer('user_id').notNull().references(() => users.id),
  teamId: integer('team_id').notNull().references(() => teams.id),
  role: varchar('role', { length: 20 }).default('member').notNull(),
  joinedAt: timestamp('joined_at').defaultNow().notNull(),
}, (table) => [
  primaryKey({ columns: [table.userId, table.teamId] }),
]);
```

## Relations

Relations are **separate from the table definition**. They enable the relational query API (`db.query.*`) but do NOT create foreign keys in the database.

### One-to-Many

```typescript
import { relations } from 'drizzle-orm';

// Parent side: one user has many posts
export const usersRelations = relations(users, ({ many }) => ({
  posts: many(posts),
}));

// Child side: one post belongs to one user
export const postsRelations = relations(posts, ({ one }) => ({
  author: one(users, {
    fields: [posts.authorId],
    references: [users.id],
  }),
}));
```

### One-to-One

```typescript
export const usersRelations = relations(users, ({ one }) => ({
  profile: one(profiles, {
    fields: [users.id],
    references: [profiles.userId],
  }),
}));

export const profilesRelations = relations(profiles, ({ one }) => ({
  user: one(users, {
    fields: [profiles.userId],
    references: [users.id],
  }),
}));
```

### Many-to-Many (via junction table)

```typescript
// Junction table
export const teamMembers = pgTable('team_members', {
  userId: integer('user_id').notNull().references(() => users.id),
  teamId: integer('team_id').notNull().references(() => teams.id),
}, (table) => [
  primaryKey({ columns: [table.userId, table.teamId] }),
]);

// Relations
export const usersRelations = relations(users, ({ many }) => ({
  teamMembers: many(teamMembers),
}));

export const teamsRelations = relations(teams, ({ many }) => ({
  teamMembers: many(teamMembers),
}));

export const teamMembersRelations = relations(teamMembers, ({ one }) => ({
  user: one(users, {
    fields: [teamMembers.userId],
    references: [users.id],
  }),
  team: one(teams, {
    fields: [teamMembers.teamId],
    references: [teams.id],
  }),
}));
```

### Self-referencing Relations

```typescript
export const categories = pgTable('categories', {
  id: integer('id').primaryKey().generatedAlwaysAsIdentity(),
  name: varchar('name', { length: 100 }).notNull(),
  parentId: integer('parent_id'),
});

export const categoriesRelations = relations(categories, ({ one, many }) => ({
  parent: one(categories, {
    fields: [categories.parentId],
    references: [categories.id],
    relationName: 'parent_child',
  }),
  children: many(categories, { relationName: 'parent_child' }),
}));
```

**Key rule:** Self-referencing and ambiguous relations require `relationName` to disambiguate.

## Schema Organization (Multi-file)

For larger projects, split schema across files:

```
src/db/
├── schema/
│   ├── index.ts      # Re-exports everything
│   ├── users.ts      # users table + usersRelations
│   ├── posts.ts      # posts table + postsRelations
│   └── teams.ts      # teams, teamMembers + their relations
├── db.ts             # drizzle() client
└── migrate.ts        # migration runner
```

The barrel export (`schema/index.ts`):

```typescript
export * from './users';
export * from './posts';
export * from './teams';
```

In `drizzle.config.ts`, point to the barrel or use a glob:

```typescript
export default defineConfig({
  schema: './src/db/schema/index.ts',
  // or: schema: './src/db/schema/*.ts',
});
```
